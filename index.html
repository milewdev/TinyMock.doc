TinyMockJS

TinyMockJS is a very small coffeescript mocking framework: 

describe "Chat", ->
  describe "greet()", ->
    it "sends a greeting and returns the response", ->
{% highlight coffeescript %}
      mock (socket) ->
        socket.expects("write").args("Hello, my name is Test\n")
        socket.expects("read").returns("Greetings Test!\n")
{% endhighlight %}        chat = new Chat(socket)
        chat.greet("Test").should.equal("Greetings Test!\n")
